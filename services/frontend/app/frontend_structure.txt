

frontend/

│   ├── hooks/              # Custom React hooks (e.g., useAuth)
│   ├── services/           # API call logic (Axios/Fetch instances)
│   ├── store/              # State management (if using Redux/Zustand)
│   ├── App.jsx             # Main Router (Connects all features)
│   ├── main.jsx            # Entry point
│   └── index.css           # Global reset/tailwind styles

├── .gitignore              # Files to skip when pushing to Git
├── Dockerfile              # The Dev/Prod Docker instructions
